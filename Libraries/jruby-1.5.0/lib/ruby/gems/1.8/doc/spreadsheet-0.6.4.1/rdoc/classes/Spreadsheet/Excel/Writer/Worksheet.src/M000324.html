  def write_multiples row, idx, multiples
    case multiples.last
    when NilClass
      write_mulblank row, idx, multiples
    when Numeric
      if multiples.size > 1
        write_mulrk row, idx, multiples
      else
        write_rk row, idx
      end
    end
  end
