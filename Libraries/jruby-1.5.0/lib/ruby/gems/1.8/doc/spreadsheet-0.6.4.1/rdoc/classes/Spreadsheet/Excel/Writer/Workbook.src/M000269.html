  def write_encoding workbook, writer
    enc = workbook.encoding || 'UTF-16LE'
    if RUBY_VERSION >= '1.9' && enc.is_a?(Encoding)
      enc = enc.name.upcase
    end
    cp = SEGAPEDOC[enc] or raise "Invalid or Unknown Codepage '#{enc}'"
    write_op writer, 0x0042, [cp].pack('v')
  end
