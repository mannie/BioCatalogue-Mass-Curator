  def content
    @content or begin
      data = nil
      data = @ole[@offset, @available]
      content, _ = @reader.read_string_body data, @flags, @available, @wide
      @continuations.each do |offset, len|
        @reader.continue_string(@ole[offset,len], [content, @chars])
      end
      content = client content, 'UTF-16LE'
      if @reader.memoize?
        @content = content
      end
      content
    end
  end
