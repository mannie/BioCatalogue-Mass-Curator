  def xf_borders
    border  = left
    border |= right       <<  4
    border |= top         <<  8
    border |= bottom      << 12
    border |= left_color  << 16
    border |= right_color << 23
    border |= cross_down  << 30
    border |= cross_up    << 31
    border
  end
