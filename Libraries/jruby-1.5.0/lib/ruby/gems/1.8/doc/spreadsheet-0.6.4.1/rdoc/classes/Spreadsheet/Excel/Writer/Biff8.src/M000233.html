  def compress_unicode_string data
    compressed = ''
    expect_null = false
    data.each_byte do |byte|
      if expect_null
        if byte != 0
          return [data, 1] # 1 => Data consists of wide Chars
        end
        expect_null = false
      else
        compressed << byte
        expect_null = true
      end
    end
    [compressed, 0] # 0 => Data consists of compressed Chars
  end
