  def xf_rotation
    rot = @format.rotation
    if @format.rotation_stacked?
      rot = 255
    elsif rot >= -90 or rotation <= 90
      rot = -rot + 90 if rot < 0
    else
      warn "rotation outside -90..90; rotation set to 0"
      rot = 0
    end
    rot
  end
