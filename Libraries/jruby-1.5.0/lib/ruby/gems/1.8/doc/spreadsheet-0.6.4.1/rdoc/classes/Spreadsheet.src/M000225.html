    def open io_or_path, mode="rb+", &block
      if io_or_path.respond_to? :seek
        Excel::Workbook.open(io_or_path)
      elsif block
        File.open(io_or_path, mode) do |fh|
          block.call open(fh)
        end
      else
        open File.open(io_or_path, mode)
      end
    end
