  def colors *args
    args.each do |key|
      attr_reader key
      define_method "#{key}=" do |name|
        name = name.to_s.downcase.to_sym
        if COLORS.include?(name)
          instance_variable_set ivar_name(key), name
        else
          raise ArgumentError, "unknown color '#{name}'"
        end
      end
    end
  end
