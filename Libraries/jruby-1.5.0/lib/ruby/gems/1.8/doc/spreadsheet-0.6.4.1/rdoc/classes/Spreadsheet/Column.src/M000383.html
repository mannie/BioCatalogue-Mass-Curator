      def updater *keys
        keys.each do |key|
          unless instance_methods.include? "unupdated_#{key}="
            alias_method "unupdated_#{key}=""unupdated_#{key}=", "#{key}=""#{key}="
            define_method "#{key}=" do |value|
              send "unupdated_#{key}=", value
              @worksheet.column_updated @idx, self if @worksheet
              value
            end
          end
        end
      end
