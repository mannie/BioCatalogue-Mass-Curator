      def method_missing name, *args, &block
        if name.to_s =~ /(.*)=$/
          return super unless args.length == 1
          return super unless PROPERTY_MAP[$1]
          self[$1] = args.first
        else
          return super unless args.length == 0
          return super unless PROPERTY_MAP[name.to_s]
          self[name]
        end
      end
