      def open mode='r'
        raise Errno::EISDIR unless file?
        io = RangesIOMigrateable.new self, mode
        @modify_time = Types::FileTime.now if io.mode.writeable?
        if block_given?
          begin   yield io
          ensure; io.close
          end
        else io
        end
      end
