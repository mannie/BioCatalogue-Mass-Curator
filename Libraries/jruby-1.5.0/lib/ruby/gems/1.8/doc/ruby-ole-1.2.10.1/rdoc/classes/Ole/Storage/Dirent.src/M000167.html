      def flatten dirents=[]
        @idx = dirents.length
        dirents << self
        if file?
          self.prev = self.next = self.child = EOT
        else
          children.each { |child| child.flatten dirents } 
          self.child = Dirent.flatten_helper children
        end
        dirents
      end
