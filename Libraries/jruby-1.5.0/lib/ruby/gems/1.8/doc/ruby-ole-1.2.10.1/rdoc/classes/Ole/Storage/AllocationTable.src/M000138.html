      def to_s
        table = truncate
        # pad it out some
        num = @ole.bbat.block_size / 4
        # do you really use AVAIL? they probably extend past end of file, and may shortly
        # be used for the bat. not really good.
        table += [AVAIL] * (num - (table.length % num)) if (table.length % num) != 0
        table.pack 'V*'
      end
